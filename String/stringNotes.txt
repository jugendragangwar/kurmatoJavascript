=====> String Metods in js

> String.charAt(pass a number)
const gameName= new String('hillclimbing');
console.log(gameName.charAt(4));

> String.length
const gameName= new String('hillclimbing');
console.log(gameName.length);

> String.toUpperCase()
const gameName= new String('hillclimbing');
console.log(gameName.toUpperCase());

> String.toLowerCase()
const gameName= new String('HillClimbinG');
console.log(gameName.toLowerCase());

> String.indexof( pass a charcter)
const gameName= new String('HillClimbinG');
console.log(gameName.indexOf('C'));

> String substring()
const gameName= new String('HillClimbinG');
console.log(gameName.substring(1,4));

> String.charCodeAt()
const gameName= new String('HillClimbinG');
console.log(gameName.charCodeAt(08));

> String at()
const gameName= new String('HillClimbinG');
console.log(gameName.at(08));

> String slice()
const gameName= new String("Apple,Mango,Grapes");
console.log(gameName.slice(6,11));

> String substr()
const gameName= new String("Apple,Mango,Grapes");
console.log(gameName.substr(0,5));
